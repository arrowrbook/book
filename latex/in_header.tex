% Custom LaTeX commands for greyscale callouts

% Define the greyscale colors
\usepackage{xcolor}
\definecolor{calloutbg}{gray}{0.9}   % Light grey background for callout boxes
\definecolor{calloutborder}{gray}{0.5}  % Grey border
\definecolor{callouttext}{gray}{0.2}    % Dark grey text

% Redefine the callout box environment to use these greyscale colors
\renewenvironment{Shaded}
{
  \begin{tcolorbox}[
    colback=calloutbg,
    colframe=calloutborder,
    coltext=callouttext,
    boxrule=0.5pt,
    left=5pt,
    right=5pt,
    top=5pt,
    bottom=5pt,
    sharp corners
  ]
}
{
  \end{tcolorbox}
}
